    <script>
        // Language switcher
        document.querySelectorAll('button[data-lang]').forEach(button => {
            button.addEventListener('click', function() {
                const lang = this.getAttribute('data-lang');
                document.querySelectorAll('[data-en]').forEach(el => {
                    el.textContent = el.getAttribute(`data-${lang}`);
                });
            });
        });

        // Modal functionality
        function openModal(src) {
            document.getElementById('modal-img').src = src;
            document.getElementById('myModal').style.display = "block";
        }

        function closeModal() {
            document.getElementById('myModal').style.display = "none";
        }

        // Close modal when clicking outside of the image
        window.addEventListener('click', function(event) {
            if (event.target === document.getElementById('myModal')) {
                closeModal();
            }
        });

        // JavaScript for rotating letters
        const letters = document.querySelectorAll('.rotating-letters span');
        const totalLetters = letters.length;
        const angle = 360 / totalLetters;

        letters.forEach((letter, index) => {
            letter.style.transform = `rotate(${angle * index}deg) translate(200px) rotate(-${angle * index}deg)`;
            letter.style.opacity = '1';
        });
    </script>
